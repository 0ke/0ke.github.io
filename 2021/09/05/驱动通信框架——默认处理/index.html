应用与驱动的通信框架

首先从应用层（ring3）进行API调用，通常情况下从动态链接库导出（kernel 32.dll）。kernel 32.dll并没有进入内核，而去寻找ntdll.dll。之后调用相应的版本。

例如：在调用OpenProcess时，ntdll.dll会去寻找NtOpenProcess或者ZwOpenProcess。

以上成为IRP结构，我们要做的就是IPR结构与驱动进行通信。即用户空间与内核空间的通信。

科普

设备：软件设备，硬件设备。

驱动：软件驱动，硬件驱动。

软件驱动：主动防御，火绒，360。

硬件驱动：键盘驱动，鼠标驱动。网卡，显卡等。

与设备进行通信的API：CreatFile（获取设备句柄 IRP_MJ_CREATE） ReadFile（去设备上取回数据 IRP_MJ_READ） WriteFile（向设备发送数据 IRP_MJ_WRITE） CloseHandle（清理工作 IRP_MJ_CLOSE）。

控制IRP：DeviceIoControl (IRP_MJ_DEVICE_CONTROL)

控制码：消息码，可以在R3和R0访问同一组控制码，例如可以在R0访问R3的控制码，来了解R3的请求。？？好像是这么回事儿，不对我看完书再改。

R0程序

#include <ntifs.h>

#define DEVICE_NAME L"\\device\\MyDriver5"
#define LINK_NAME L"\\dosdevices\\0ke"
#define IRP_IOCTRL_BASE 0x8000        //控制码的起始位置，起始位置0x8000
#define IRP_IOCTRL_CODE(i) CTL_CODE(FILE_DEVICE_UNKNOWN,IRP_IOCTRL_BASE + i,METHOD_BUFFERED,FILE_ANY_ACCESS)//控制码的宏定义
#define CTL_PRINT IRP_IOCTRL_CODE(1)//控制码定义

//IRP派遣函数--默认处理
NTSTATUS DispatchCommon(PDEVICE_OBJECT pDeviceObject, PIRP pIrp)
{
    //DbgPrint("Common!");
    pIrp->IoStatus.Status = STATUS_SUCCESS;    //设置IRP处理成功
    pIrp->IoStatus.Information = 0;            //设置返回的字节数，ReadFile能读多少个字节，设置成0，返回随机数，不想设置
    IoCompleteRequest(pIrp, IO_NO_INCREMENT);    //结束IRP处理流程

    return STATUS_SUCCESS;            //返回成功
}

//ReadFile
NTSTATUS DispatchRead(PDEVICE_OBJECT pDeviceObject, PIRP pIrp)
{

    ULONG ulMin = 0;                    
    PVOID pReadBuffer = NULL;    //缓冲区指针，初始化为NULL
    ULONG ulReadLength = 0;        //ULONG保存缓冲区长度，初始化为0
    PIO_STACK_LOCATION pStack = NULL;    //IRP栈结构指针，初始化为NULL
    pReadBuffer = pIrp->AssociatedIrp.SystemBuffer;        //获取缓冲区位置
    pStack = IoGetCurrentIrpStackLocation(pIrp);        //获取当前层IRP栈结构
    ulReadLength = pStack->Parameters.Read.Length;        //获取读写长度
    ULONG ulStrLength = ((wcslen(L"Hello World!") + 1) * sizeof(WCHAR));    //计算实际字符串长度
    ulMin = ulReadLength > ulStrLength ? ulStrLength : ulReadLength;        
    RtlCopyMemory(pReadBuffer, L"Hello World!", ulMin);                //拷贝数据
    pIrp->IoStatus.Status = STATUS_SUCCESS;//设置IRP处理成功
    pIrp->IoStatus.Information = ulMin;        //拷多少数据，传多少数据
    IoCompleteRequest(pIrp, IO_NO_INCREMENT);//结束IRP处理流程
    //DbgPrint("Driver Load sucess");
    return STATUS_SUCCESS;
}


NTSTATUS DispatchWrite(PDEVICE_OBJECT pDeviceObject, PIRP pIrp)
{
    //DbgBreakPoint();                    //断点调试
    PVOID pReadBuffer = NULL;            //写入缓冲区指针    
    ULONG ulReadLength = 0;                //写入长度。初始化为0，这里需要注意，R3程序未必会按照R0设定好的长度发送
    PIO_STACK_LOCATION pStack = NULL;    //IRP栈指针
    pReadBuffer = pIrp->AssociatedIrp.SystemBuffer;//这里pReadBuffer存放的已经是R3传过来的数据了
    pStack = IoGetCurrentIrpStackLocation(pIrp);//获取当前IRP栈结构
    ulReadLength = pStack->Parameters.Write.Length;//获取写入长度
    //DbgPrint("%ws",pWriteBuffer);
    UNICODE_STRING usStr = { 0 };
    RtlInitUnicodeString(&usStr, pReadBuffer);
    DbgPrint("%wZ\r\n", usStr);
    pIrp->IoStatus.Status = STATUS_SUCCESS;    //设置IRP处理成功
    pIrp->IoStatus.Information = ulReadLength;//设置返回字节数
    IoCompleteRequest(pIrp, IO_NO_INCREMENT);//结束IRP处理流程
    return STATUS_SUCCESS;//返回成功
}
//IRP_MJ_DEVICE_CONTROL
NTSTATUS DispatchIoCtrl(PDEVICE_OBJECT pDeviceObject, PIRP pIrp)
{
    //DbgBreakPoint();
    ULONG ulIoctrlCode = 0;    //定义控制码变量，初始化为0
    PVOID pSystemBuffer = NULL;    //输入缓冲区，这里注意，可以设置两个缓冲区。
    ULONG ulBufferLength = 0;//输入缓冲区长度。
    PIO_STACK_LOCATION pStack = NULL;//IRP栈指针，初始化指向NULL
    pSystemBuffer = pIrp->AssociatedIrp.SystemBuffer;//获取缓冲区。指向SystemBuffer
    pStack = IoGetCurrentIrpStackLocation(pIrp);//获取当前层IRP栈
    ulBufferLength = pStack->Parameters.DeviceIoControl.InputBufferLength;//获取输入缓冲区长度，从R3过来的数据有多长。使用DeviceIoControl中的InputBufferLength
    ulIoctrlCode = pStack->Parameters.DeviceIoControl.IoControlCode;//获取控制码，也是使用DeviceIoControl结构中的内容。
    switch (ulIoctrlCode)//按照控制码分发，switch处理消息很好用。
    {
    case CTL_PRINT;//
    {
        DbgPrint("%s\r\n", pSystemBuffer);//打印R3来的数据。存储在pSystemBuffer，宽字符就是%ws,普通字符是%s
        ULONG uRetLength = strlen("io success 0ke!");//获取字符串长度
        RtlZeroMemory(pSystemBuffer, 1024);//初始化R3的内存.
        RtlCopyMemory(pSystemBuffer, "io success 0ke!", uRetLength);//内存拷贝。
        pIrp->IoStatus.Status = STATUS_SUCCESS;//底下这堆和以前一样。
        pIrp->IoStatus.Information = uRetLength;//返回多少个字节
        IoCompleteRequest(pIrp, IO_NO_INCREMENT);//结束IRP处理流程
        return STATUS_SUCCESS;//返回成功
    }
    default:
        break;
    }
}

NTSTATUS DriverEntry(PDRIVER_OBJECT pDriverObject, PUNICODE_STRING pRegPath)
{
    UNICODE_STRING usDrviceName = { 0 };  //设备对象
    UNICODE_STRING usLinkName = { 0 };        //符号链接名称
    NTSTATUS ntStatus = STATUS_UNSUCCESSFUL;
    RtlInitUnicodeString(&usDrviceName, DEVICE_NAME);    //初始化设备对象
    RtlInitUnicodeString(&usLinkName, LINK_NAME);        //初始化符号链接
    //创建设备对象
    PDEVICE_OBJECT pDeviceObject = NULL;
    ntStatus = IoCreateDevice(pDriverObject, 0, &usDrviceName,FILE_DEVICE_UNKNOWN/*不知道啥类型的设备*/, 0, TRUE, &pDeviceObject);
    if (!NT_SUCCESS(ntStatus))//判断设备对象是否创建成功
    {
        DbgPrint("IoCreateDevice Failed!");
        return ntStatus;
    }
    pDeviceObject->Flags |= DO_BUFFERED_IO;        //基于缓存的IO方式
    //IO方式有很多种
    //基于缓存的IO方式
    //直接读写的IO方式
    //两者皆不的IO方式，具体使用哪种IO方式，改变pDeviceObject->Flags。
    ntStatus = IoCreateSymbolicLink(&usLinkName, &usDrviceName);//创建符号链接
    if (!NT_SUCCESS(ntStatus))//判断符号连接是否成功
    {
        //IoDeleteDevice(pDeciceObject);   //删除设备对象
        DbgPrint("IoCreateSymbolicLink Failed!");
        return ntStatus;
    }

    for (size_t i = 0; i < IRP_MJ_MAXIMUM_FUNCTION+1/*一个总数宏，加1是因为他从0开始计数*/; i++)//将所有
    {
        pDriverObject->MajorFunction[i] = DispatchCommon;//驱动对象中的派遣函数，循环都指定成默认处理
    }
    pDriverObject->MajorFunction[IRP_MJ_READ] = DispatchRead;    //指定处理Read
    pDriverObject->MajorFunction[IRP_MJ_WRITE] = DispatchWrite;//指定处理write
    pDriverObject->MajorFunction[IRP_MJ_DEVICE_CONTROL] = DispatchIoCtrl;//指定处理Iocontrol
    pDriverObject->DriverUnload = DriverUnload;//注册卸载函数
    return STATUS_SUCCESS;
}

R3程序

#include <stdio.h>
#include <Windows.h>

#define LINK_NAME L"\\\\.\\0ke"        //符号链接
#define IRP_IOCTRL_BASE 0x8000
#define IRP_IOCTRL_CODE(i) CTL_CODE(FILE_DEVICE_UNKNOWN,IRP_IOCTRL_BASE + i,METHOD_BUFFERED,FILE_ANY_ACCESS)
#define CTL_PRINT IRP_IOCTRL_CODE(1)
int main()
{
    //IRP_MJ_DEVICE_CONTROL
    HANDLE hFile = CreateFile(LINK_NAME, GENERIC_ALL, 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);            //获取设备句柄并接收
    if (hFile == INVALID_HANDLE_VALUE)//判断是否打开成功
    {
        printf("CreateFile Failed\n");
        system("pause");
        return 1;
    }

    while (true)
    {
        CHAR szBufferInput[1024] = { 0 };
        CHAR szBuuferOutput[1024] = { 0 };
        printf("请输入字符串:");
        scanf("%s", &szBufferInput);
        DWORD dwRet = 0;
        BOOL bRet = DeviceIoControl(hFile,
            CTL_PRINT,
            szBufferInput,
            sizeof(szBufferInput),
            szBuuferOutput,
            sizeof(szBuuferOutput),
            &dwRet,
            NULL);//获取创建句柄de
        if (bRet)
        {
            printf("Output = %s\r\n", szBuuferOutput);//返回什么内容
            printf("nRet = %d\r\n", dwRet);//返回多少个字节
        }
        else
        {
            printf("io failed\r\n", szBuuferOutput);//判断返回失败
        }

    }


    UCHAR buffer[MAX_PATH];            //MAX_PATH代表256字节。
    ULONG ulRead;
    ReadFile(hFile, buffer, MAX_PATH, &ulRead, NULL);    //读取
    buffer[ulRead * 2 + 1] = '\0';
    wprintf(L"%s", buffer);
    //IRP_MJ_WRITE
    ULONG ulWrite;
    DWORD dwWriteLength = wcslen(L"Hello World 0ke") * sizeof(WCHAR) + 1;    
    WriteFile(hFile, L"Hello World 0ke", dwWriteLength, &ulWrite, NULL);    //dwWriteLength期望写的数据，&ulWrite实际写的数据
    CloseHandle(hFile);
    system("pause");
    return 0;
}
